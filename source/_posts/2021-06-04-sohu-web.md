- 自己实现一个 bind 函数

```js
Function.prototype.bind = function (obj, ...arg) {
  var context = this;
  var bound = function (...newArg) {
    arg = arg.concat(newArg);
    return context.apply(obj, arg);
  };
  var F = function() {};
  F.prototype = context.prototype;
  bound.prototype = new F();
  return bound;
};

function func(a, b) {
  console.log(this);
  console.log(a);
  console.log(b);
}

const newFunc = func.bind({ new: true }, 1);
newFunc(2);
```
